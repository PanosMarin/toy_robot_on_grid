 42class PlaceRobotCommand(RobotCommand):
 43    '''Implements a robot place command'''
 44
 45    def __init__(self, coordinates: Coordinates, direction: Direction):
 46        '''
 47        Initialises a robot placement command.
 48        '''
 49        self._coordinates = coordinates
 50        self._direction = direction

 52    def execute(self, robot: 'Robot'):
 53        '''
 54        Places the robot on the grid.
 55
 56        Args:
 57            robot(Robot): Robot to execute the command.
 58        Returns:
 59            bool: True if operation was successful, False otherwise.
 60        '''
 61        if robot.get_grid().is_valid_cell(self._coordinates):
 62            robot.coordinates = self._coordinates
 63            robot.direction = self._direction
 64            return True
 65        else:
 66            print(f'Cell with coordinates {self._coordinates} not valid.')
 67            return False

158class RobotApiHandler(AbastractRobotApiHandler):
159    '''Class that generates robot commands from string commands.'''
160    initialised = False
161
162    @classmethod
163    def _initialise(cls):
164        cls.initialised = True
165        cls.api_generator_map = {
166            'PLACE': cls._generate_place_command,
167            'MOVE': cls._generate_move_command,
168            'LEFT': cls._generate_turn_command,
169            'RIGHT': cls._generate_turn_command,
170            'REPORT': cls._generate_report_command
171        }
172
173    @classmethod
174    def generate_command(cls, command: str) -> RobotCommand:
175        '''
176        Generates robot command based on API specifics.
177        Args:
178            data(str): Data specific to API for command generation.
179        Returns:
180            RobotCommand: Robot command generated.    
181        '''
182        if not cls.initialised:
183            cls._initialise()
184        try:
185            data = command.split(' ')
186            command_generator = cls.api_generator_map.get(data[0], None)
187            if command_generator is None:
188                raise RuntimeError('Not valid command type.')
189            return command_generator(data)
190        except Exception as e:
191            print(f'Invalid command format. Exception {e}. Command {command}')
192            return NopRobotCommand()
193
194    @classmethod
195    def _generate_place_command(cls, data: List[str]) -> PlaceRobotCommand:
196        '''
197        Generates a robot place command.
198
199        Returns:
200            PlaceRobotCommand: Returns a Place robot command with the specified parameters.
201        Raises:
202            Runtime error if command has invalid format.
203            Value error if any of the parameters are not of the expected type.
204        '''
205        if len(data) != 2:
206            raise RuntimeError('Invalid place command format.')
207        params = data[1].split(',')
208
209        if not params[0].isdigit():
210            raise ValueError('X coordinate in not a digit.')
211        if not params[1].isdigit():
212            raise ValueError('X coordinate in not a digit.')
213        if params[2] not in Direction.__members__:
214            raise ValueError('Direction provided in command not implemented.')
215
216        return PlaceRobotCommand(Coordinates(int(params[0]), int(params[1])),
217                                 Direction[params[2]])
218
219    @classmethod
220    def _generate_turn_command(cls, data: List[str]) -> TurnRobotCommand:
221        '''
222        Generates a robot turn command.
223
224        Returns:
225            TurnRobotCommand: Returns a Turn robot command with the specified parameters.
226        Raises:
227            Runtime error if command has invalid format.
228            Value error if any of the parameters are not of the expected type.
229        '''
230        if len(data) != 1:
231            raise RuntimeError('Invalid turn command format.')
232        if data[0] not in RobotRotation.__members__:
233            raise RuntimeError('Invalid robot rotation command.')
234        return TurnRobotCommand(RobotRotation[data[0]])

245            Value error if any of the parameters are not of the expected type.

247        if len(data) != 1:

279class StubFileCommandReceiver(AbstractCommandReceiver):
280    '''
281    Abstract class for command receiver.
282    '''
283
284    def __init__(
285            self,
286            file: str,
287            api_handler: AbastractRobotApiHandler = RobotApiHandler):
288        with open(file, 'r') as f:
289            self._commands: list[str] = f.readlines()
290            self._api_handler = api_handler()
291
292    def recv(self) -> RobotCommand:
293        '''
294        Returns:
295            RobotCommand: Returns the robot command received.
296        '''
297        while True:
298            if len(self._commands) < 1:
299                return None
300            command = self._api_handler.generate_command(
301                self._commands.pop(0).strip('\n'))
302            if command is None:
303                continue
304            return command

307class Robot:
308    '''Class that implements a robot on a grid like structure'''
309
310    def __init__(
311            self,
312            grid: AbstractGrid,
313            command_receiver: AbstractCommandReceiver):
314        '''
315        Initialises the toy robot.
316
317        Args:
318            grid(AbstractGrid): Grid for the robo to to be placed.
319        '''
320        self._coordinates: Coordinates = None
321        self._direction: Direction = None
322        self._grid = grid
323        self._command_receiver = command_receiver
324
325    @property
326    def direction(self) -> Direction:
327        '''Get direction'''
328        return self._direction
329
330    @direction.setter
331    def direction(self, direction) -> None:
332        """Set the direction."""
333        self._direction = direction
334
335    @property
336    def coordinates(self):
337        '''Get coordinates'''
338        if self._coordinates is None:
339            return None
340        return Coordinates(self._coordinates.x, self._coordinates.y)
341
342    @coordinates.setter
343    def coordinates(self, coordinates: Coordinates):
344        '''Set coordinates'''
345        self._coordinates = Coordinates(coordinates.x, coordinates.y)
346
347    def is_on_grid(self) -> bool:
348        '''
349        Checks if a robot has a valid grid position
350
351        Returns:
352            bool: Returns true if robot has been placed on the grid.
353        '''
354        return self._coordinates is not None and self._direction is not None
355
356    def get_grid(self) -> AbstractGrid:
357        '''Get grid'''
358        return self._grid
359
360    def spin(self):
361        '''Spin robot to run through the commands. Exits when no more commands to be received.'''
362        while True:
363            command = self._command_receiver.recv()
364            if command is None:
365                break
366            command.execute(self)
367        print('No more commands to execute...')
